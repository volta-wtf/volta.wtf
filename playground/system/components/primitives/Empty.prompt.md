    <EmptyMedia><Icon name="inbox" /></EmptyMedia>

    <EmptyDescription>Create your first project to get started.</EmptyDescription>
  </EmptyHeader>
  <EmptyContent><Button><Icon name="add" /> New project</Button></EmptyContent>
</Empty>
